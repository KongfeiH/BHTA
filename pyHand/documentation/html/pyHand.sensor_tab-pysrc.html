 75          @type fill: C{String}

 83          def __init__(self, canvas, coord=(0,0),offset=(0,0),width=0, 
 84                       length=0,fill="",stipple="",text="0"): 
 85              self.coord=coord 
 86              self.offset=offset 
 87              self.width=width 
 88              self.length=length 
 89              self.fill=fill 
 90              self.stipple=stipple 
 91              self.text=text

 93              self.make_rect(canvas)

 95          def make_rect(self, canvas): 
 96              ''' 
 97              Creates a rectangle object on a canvas. 
 98   
 99              @param canvas: canvas rectangle is to be created on. 
100              ''' 
101              self.rect=canvas.create_rectangle( 
102                        self.coord, 
103                        self.coord[0]+self.width, 
104                        self.coord[1]+self.length, 
105                        outline="black", 
106                        fill=self.fill, 
107                        stipple=self.stipple) 
108              canvas.move(self.rect, self.offset[0], self.offset[1]) 
109              self.label= canvas.create_text(zero(self.coord[0], self.coord[1]), 
110                                             text=self.text)

136      def zero(x, y): 
137          ''' 
138          Zeros out coordinates of a canvas object since (0,0) on canvas is not in top left corner. 
139   
140          @param x: x vlaue of position coordinate (can be int or float). 
141          @param y: y value of position coordinate (can be int or float). 
142   
143          @return offsetx: the new x value of position offset with top left corner being (0,0). 
144          @return offsety: the new y value of position offset with top left corner being (0,0). 
145          ''' 
146          offsetx=0 
147          offsety=0 
148          if x>=0: 
149              offsetx= -274+x 
150              offsety= -193+y 
151          return offsetx, offsety

260          ''' 
261          A single recursive function call that updates the values of each tactile sensing unit as well as the 
262          puck temp, motor temp, and fingertip torque sensors. This ws done in a single recursive call to eliminate 
263          time dealys for when  multiple processes occurred simultaniously.

265          @param canvas: the canvas the tactile unit rectangle is created on. 
266   
267          @param on_off: the start tactile array sensor variable which indicates the state of the button (1 is on, 0 is off)

270          @param matrix: an array of fingers (and palm) which consist of array of tact_units.  
271          @type matrix: C{[[tact_unit]]} 
272   
273          @param s: HandSensor object 
274   
275          @param label1: fingertip torque sensor label for Finger1 
276          @type label1: C{Label_Widget} 
277   
278          @param label2: fingertip torque sensor label for Finger2 
279          @type label1: C{Label_Widget} 
280   
281          @param label3: fingertip torque sensor label for Finger3 
282          @type label1: C{Label_Widget} 
283   
284          @param puck_labels: puck temperature label for all fingers 
285          @type label1: C{[Label_Widget]} 
286   
287          @param motor_labels: motor temperature label for all fingers 
288          @type label1: C{[Label_Widget]} 
289   
290          @param sat_label_puck:  The saturation warning label for puck temperature 
291          @type label1: C{Label_Widget} 
292   
293          @param sat_label_motor: The saturation warning label for motor temperature 
294          @type label1: C{Label_Widget} 
295   
296          @param sat_label_strain: The saturation warning label for fingertip torque sensor 
297          @type label1: C{Label_Widget} 
298   
299          @param counter: parameter to keep track of update/refresh times (int of 0-100) 
300   
301          @param nb: notebook to enable tabs from. 
302           
303          ''' 
304          update_rate=10 
305          if nb.select()!="": 
306              if nb.tab(nb.select(), "text") == "Sensors": 
307                  counter+=1 
308                  if counter>100: 
309                      counter=0 
310   
311                  if on_off.get()==1: 
312                      s.get_full_tact() 
313                       
314                      arr=matrix[0]  
315                      for i in range(len(s.finger1)): 
316                          canvas.itemconfig(arr[i].label,text=s.finger1[i]) 
317                          update_color(canvas,arr[i].rect,s.finger1[i]) 
318   
319                      arr2=matrix[1]  
320                      for j in range(len(s.finger2)): 
321                          canvas.itemconfig(arr2[j].label,text=s.finger2[j]) 
322                          update_color(canvas,arr2[j].rect,s.finger2[j]) 
323   
324                      arr3=matrix[2]  
325                      for i in range(len(s.finger3)): 
326                           canvas.itemconfig(arr3[i].label,text=s.finger3[i]) 
327                           update_color(canvas,arr3[i].rect,s.finger3[i])         
328   
329                      arr4=matrix[3] 
330                      for i in range(len(s.spread)): 
331                           canvas.itemconfig(arr4[i].label,text=s.spread[i]) 
332                           update_color(canvas,arr4[i].rect,s.spread[i]) 
333                        
334                  if on_off.get()==1:      
335                      if counter%10==0: 
336                          on_strain_call(w,FINGER1,sat_label_strain, strain) 
337                      if (counter+4)%10==0: 
338                          on_strain_call(w,FINGER2,sat_label_strain, strain) 
339                      if (counter+8)%10==0: 
340                          on_strain_call(w,FINGER3,sat_label_strain, strain) 
341   
342                      if counter%100==0: 
343                          #Puck Temperature 
344                          on_puck_temp_call(puck_labels[0],FINGER1,sat_label_puck, handtype) 
345                      if (counter+12)%100==0: 
346                          on_puck_temp_call(puck_labels[1],FINGER2,sat_label_puck,handtype) 
347                      if (counter+24)%100==0: 
348                          on_puck_temp_call(puck_labels[2],FINGER3,sat_label_puck,handtype)

360                          on_motor_temp_call(motor_labels[3],SPREAD,sat_label_motor,thermister) 
361                           
362                  if on_off.get()==0: 
363                      on_strain_call(w,FINGER1,sat_label_strain, strain)

365                      on_strain_call(w,FINGER3,sat_label_strain, strain)

703      def strain_to_nm(x): 
704          ''' 
705          Converts from raw encoder unit reading to Newton-meters. 
706   
707          @param w: the canvas the fingertip torque sensor bar graph is created on. 
708   
709          @returns nm: the torque value in Newton-meters.  
710          ''' 
711          p1 = 2.754e-10 
712          p2 = -1.708e-06 
713          p3 = 0.003764 
714          p4 = -2.85 
715          nm= p1*x**3 + p2*x**2 + p3*x + p4 
716          return nm

721   
722          @param w: the canvas the fingertip torque sensor bar graph is created on.
